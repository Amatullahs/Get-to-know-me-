import numpy as np import matplotlib.pyplot as plt # Generate data points for demand curve quantity_demanded = np.arange(0, 10, 1) price_demand = 10 - quantity_demanded # Generate data points for supply curve quantity_supplied = np.arange(0, 10, 1) price_supply = quantity_supplied # Equilibrium point equilibrium_quantity = 5 equilibrium_price = 5 # Price set above equilibrium price_above = 7 # Create the plot plt.figure(figsize=(8, 6)) # Plot demand and supply curves plt.plot(quantity_demanded, price_demand, label='Demand Curve', color='blue') plt.plot(quantity_supplied, price_supply, label='Supply Curve', color='green') # Plot equilibrium point plt.scatter(equilibrium_quantity, equilibrium_price, color='red', label='Equilibrium Point') # Plot horizontal line at price above equilibrium plt.axhline(y=price_above, color='orange', linestyle='--', label='Price Above Equilibrium') # Add labels and legend plt.xlabel('Quantity of Corn') plt.ylabel('Price of Corn') plt.title('Market for Corn') plt.legend() # Show plot plt.grid(True) plt.show()
